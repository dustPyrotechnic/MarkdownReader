# MarkdownReader MVP 实施计划

> **For Claude:** REQUIRED SUB-SKILL: Use superpowers:executing-plans to implement this plan task-by-task.

**Goal:** 构建一款 iOS 18+ Markdown 阅读器，支持从微信/QQ 导入 .md 文件，完整渲染 GFM + 数学公式 + 流程图，多级文件夹管理，以及划线批注功能。

**Architecture:** MV + @Observable 架构，三层分离：SwiftData Model 描述数据，@Observable Store 处理业务逻辑，SwiftUI View 纯 UI 展示。渲染层使用 WKWebView 加载离线 JS 引擎（marked.js + KaTeX + Mermaid.js），通过 WKScriptMessageHandler 与 Swift 层双向通信。

**Tech Stack:** Swift 6.2、SwiftUI、SwiftData、WebKit（WKWebView）、marked.js、KaTeX、Mermaid.js、highlight.js、SWCharSphere（MIT）

**Files:**

**Step 1: 创建 FolderStore.swift**

```swift
import SwiftData
import Foundation

@MainActor
@Observable
final class FolderStore {
    private(set) var rootFolders: [Folder] = []

    func load(context: ModelContext) {
        let descriptor = FetchDescriptor<Folder>(
            predicate: #Predicate { $0.parent == nil },
            sortBy: [SortDescriptor(\.createdAt)]
        )
        rootFolders = (try? context.fetch(descriptor)) ?? []
    }

    func createFolder(name: String, parent: Folder? = nil, context: ModelContext) {
        let folder = Folder(name: name, parent: parent)
        context.insert(folder)
        load(context: context)
    }

    func deleteFolder(_ folder: Folder, context: ModelContext) {
        context.delete(folder)
        load(context: context)
    }

    func move(_ document: Document, to folder: Folder, context: ModelContext) {
        document.folder = folder
        load(context: context)
    }

    /// App 首次启动时确保"未分类"文件夹存在
    func ensureInbox(context: ModelContext) {
        load(context: context)
        guard !rootFolders.contains(where: { $0.name == "未分类" }) else { return }
        createFolder(name: "未分类", context: context)
    }
}
```

### Task 5：MarkdownWebView 桥接层

**Files:**
- Create: `MarkdownReader/WebView/MarkdownWebView.swift`

**Step 1: 创建 MarkdownWebView.swift**

```swift
import SwiftUI
import WebKit

struct MarkdownWebView: UIViewRepresentable {
    let markdown: String
    var onSelectionChanged: ((String) -> Void)?

    func makeCoordinator() -> Coordinator {
        Coordinator(onSelectionChanged: onSelectionChanged)
    }

    func makeUIView(context: Context) -> WKWebView {
        let config = WKWebViewConfiguration()
        config.userContentController.add(context.coordinator, name: "selectionChanged")

        let webView = WKWebView(frame: .zero, configuration: config)
        webView.scrollView.showsVerticalScrollIndicator = false
        webView.isOpaque = false
        webView.backgroundColor = .clear

        loadRenderer(in: webView)
        return webView
    }

    func updateUIView(_ webView: WKWebView, context: Context) {
        context.coordinator.pendingMarkdown = markdown
        context.coordinator.webView = webView
        if context.coordinator.isLoaded {
            renderMarkdown(markdown, in: webView)
        }
    }

    private func loadRenderer(in webView: WKWebView) {
        guard let url = Bundle.main.url(forResource: "renderer", withExtension: "html") else { return }
        webView.loadFileURL(url, allowingReadAccessTo: url.deletingLastPathComponent())
    }

    private func renderMarkdown(_ markdown: String, in webView: WKWebView) {
        let escaped = markdown
            .replacingOccurrences(of: "\\", with: "\\\\")
            .replacingOccurrences(of: "`", with: "\\`")
        webView.evaluateJavaScript("renderMarkdown(`\(escaped)`)", completionHandler: nil)
    }

    // MARK: - Coordinator

    final class Coordinator: NSObject, WKScriptMessageHandler {
        var isLoaded = false
        var pendingMarkdown = ""
        weak var webView: WKWebView?
        var onSelectionChanged: ((String) -> Void)?

        init(onSelectionChanged: ((String) -> Void)?) {
            self.onSelectionChanged = onSelectionChanged
        }

        func userContentController(
            _ userContentController: WKUserContentController,
            didReceive message: WKScriptMessage
        ) {
            guard message.name == "selectionChanged",
                  let body = message.body as? [String: Any],
                  let text = body["text"] as? String else { return }
            DispatchQueue.main.async { self.onSelectionChanged?(text) }
        }
    }
}
```

```swift
        .task { await loadMarkdown() }
    }

    private func loadMarkdown() async {
        let url = URL.documentsDirectory.appending(path: document.filePath)
        markdown = (try? String(contentsOf: url, encoding: .utf8)) ?? "# 文件读取失败"

        let words = text
            .components(separatedBy: .whitespacesAndNewlines)
            .filter { $0.count > 1 && !$0.hasPrefix("#") && !$0.hasPrefix("`") }
        return Array(Set(words.prefix(60))).map { String($0.prefix(4)) }
    }
```

```swift
import SwiftData
import Foundation

@MainActor
@Observable
final class DocumentImporter {
    var isImporting = false
    var lastError: String?

    func importFile(
        from sourceURL: URL,
        into folder: Folder,
        context: ModelContext
    ) async {
        isImporting = true
        defer { isImporting = false }

        do {
            let accessing = sourceURL.startAccessingSecurityScopedResource()
            defer { if accessing { sourceURL.stopAccessingSecurityScopedResource() } }

            let fileName = sourceURL.lastPathComponent
            let dest = URL.documentsDirectory
                .appending(path: folder.name)
                .appending(path: fileName)

            try FileManager.default.createDirectory(
                at: dest.deletingLastPathComponent(),
                withIntermediateDirectories: true
            )

            if FileManager.default.fileExists(atPath: dest.path) {
                try FileManager.default.removeItem(at: dest)
            }
            try FileManager.default.copyItem(at: sourceURL, to: dest)

            let relativePath = "\(folder.name)/\(fileName)"
            let doc = Document(fileName: fileName, filePath: relativePath, folder: folder)
            context.insert(doc)
        } catch {
            lastError = error.localizedDescription
        }
    }
}
```

```swift
import UIKit
import Social
import UniformTypeIdentifiers

final class ShareViewController: UIViewController {

    override func viewDidAppear(_ animated: Bool) {
        super.viewDidAppear(animated)
        guard let item = extensionContext?.inputItems.first as? NSExtensionItem,
              let provider = item.attachments?.first(where: {
                  $0.hasItemConformingToTypeIdentifier(UTType.text.identifier) ||
                  $0.hasItemConformingToTypeIdentifier("net.daringfireball.markdown")
              }) else {
            extensionContext?.completeRequest(returningItems: nil)
            return
        }

        let typeID = provider.hasItemConformingToTypeIdentifier("net.daringfireball.markdown")
            ? "net.daringfireball.markdown"
            : UTType.text.identifier

        provider.loadItem(forTypeIdentifier: typeID) { [weak self] item, _ in
            DispatchQueue.main.async {
                if let url = item as? URL {
                    self?.openMainApp(with: url)
                }
                self?.extensionContext?.completeRequest(returningItems: nil)
            }
        }
    }

    private func openMainApp(with url: URL) {
        guard let appURL = URL(string: "markdownreader://import?path=\(url.absoluteString)") else { return }
        var responder: UIResponder? = self
        while let r = responder {
            if let app = r as? UIApplication {
                app.open(appURL)
                return
            }
            responder = r.next
        }
    }
}
```

```swift
import SwiftData
import Foundation

@MainActor
@Observable
final class AnnotationStore {
    func add(
        text: String,
        emoji: String?,
        rangeStart: Int,
        rangeEnd: Int,
        to document: Document,
        context: ModelContext
    ) {
        let annotation = Annotation(
            rangeStart: rangeStart,
            rangeEnd: rangeEnd,
            comment: text,
            emoji: emoji
        )
        annotation.document = document
        context.insert(annotation)
    }

    func delete(_ annotation: Annotation, context: ModelContext) {
        context.delete(annotation)
    }
}
```

```swift
import SwiftUI
import SwiftData

struct AnnotationHUD: View {
    let selectedText: String
    let document: Document
    @Environment(\.modelContext) private var context
    @Environment(\.dismiss) private var dismiss
    @State private var comment = ""
    @State private var selectedEmoji: String?
    private let emojis = ["⭐️", "❓", "💡", "⚠️", "❤️", "👍", "📌", "🔥"]

    var body: some View {
        VStack(alignment: .leading, spacing: 16) {
            Text("批注")
                .bold()
                .padding(.top)

            Text(""\(selectedText.prefix(60))"")
                .font(.caption)
                .foregroundStyle(.secondary)
                .lineLimit(2)

            TextField("添加评注…", text: $comment, axis: .vertical)
                .lineLimit(2...4)
                .textFieldStyle(.roundedBorder)

            ScrollView(.horizontal, showsIndicators: false) {
                HStack {
                    ForEach(emojis, id: \.self) { emoji in
                        Button(emoji) { selectedEmoji = selectedEmoji == emoji ? nil : emoji }
                            .font(.title2)
                            .padding(6)
                            .background(selectedEmoji == emoji ? Color.accentColor.opacity(0.2) : .clear)
                            .clipShape(.rect(cornerRadius: 8))
                    }
                }
            }

            Button("保存批注") {
                let store = AnnotationStore()
                store.add(
                    text: comment,
                    emoji: selectedEmoji,
                    rangeStart: 0,
                    rangeEnd: selectedText.count,
                    to: document,
                    context: context
                )
                dismiss()
            }
            .buttonStyle(.borderedProminent)
            .disabled(comment.trimmingCharacters(in: .whitespaces).isEmpty && selectedEmoji == nil)
        }
        .padding()
    }
}
```
